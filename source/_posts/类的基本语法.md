### 属性表达式
```js
let shopping = "shop";
        class Person{
            constructor(sex) {
                this.sex = sex;
                this.hobby = "女";
            }

            [shopping](){
                console.log("一个省钱的春节");
            }
        }
       
        let person = new Person();
        person.shop(); 
```
